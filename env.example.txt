# Configuração do Banco Oracle 11g (Origem)
ORACLE_11G_DSN=host11g:1521/ORCL11G
ORACLE_11G_USER=usuario_origem
ORACLE_11G_PASSWORD=senha_origem
# Opcional: Path específico do Oracle Client para 11g (pode ser versão mais recente)
# ORACLE_11G_CLIENT_PATH=C:\oracle\instantclient_19_3

# Configuração do Banco Oracle 9i (Destino)
ORACLE_9I_DSN=host9i:1521/ORCL9I
ORACLE_9I_USER=usuario_destino
ORACLE_9I_PASSWORD=senha_destino
# OBRIGATÓRIO: Path específico do Oracle Client para 9i (deve ser 11.2 ou anterior)
# Descomente e configure o caminho correto do Oracle Instant Client 11.2:
ORACLE_9I_CLIENT_PATH=C:\oracle\instantclient_11_2
# OU use o caminho que você mencionou:
# ORACLE_9I_CLIENT_PATH=C:\ORAWIN95\bin

# Opcional: Path global do Oracle Client (usado como fallback se os paths específicos não estiverem definidos)
# ORACLE_CLIENT_PATH=C:\oracle\instantclient_11_2

# Opcional: Schema específico (se não especificado, usa o usuário como schema)
# ORACLE_11G_SCHEMA=SCHEMA_ORIGEM
# ORACLE_9I_SCHEMA=SCHEMA_DESTINO
